      DATA:
      
      struct Lift: Identifiable, Codable
        let id: UUID
        ex. name: String
        timestamp: Date
        weight: Double
        reps: Int
        tempo: [Int, Int, Int] - [eccentric, pause, concentric]
            if not provided -> default tempo
        isPR: bool
        notes: String
        perceivedEffort: Int // RPE 1-10
        
      struct Exercise: Identifiable, Codable
        let id: UUID
        name: String
        targetedMuscles: [Muscles]
        defaultTempo: [Int, Int, Int]   // Optional default tempo for the exercise
        equipment: EquipmentType?
      
      enum Muscles: String, Codable, CaseIterable {
        case chest, lats, mid back, traps, triceps, biceps...
      }
      
      enum EquipmentType: String, Codable, CaseIterable {
        case barbell, dumbbell, machine, cable, kettlebell
        case bodyweight, bands, chains, slingshot, other
      }
